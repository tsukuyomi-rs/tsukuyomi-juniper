Adds a route into the global scope.

Examples

use tsukuyomi::app::builder::Route;

fn handler(_: &mut Input) -> Handle {
    // ...
}
fn submit(_: &mut Input) -> Handle {
    // ...
}

let app = App::builder()
    .route(("/", handler))
    .route(("/", Method::POST, submit))
    .route(|r: &mut Route| {
        r.uri("/submit");
        r.method(Method::POST);
        r.handler(submit);
    })
    .finish()?;
